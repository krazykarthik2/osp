[BITS 16]
[ORG 0x7e00]

start:
    cli
    xor ax, ax
    mov ss, ax    
    mov ds, ax
    mov es, ax
    sti

    mov si, hello_msg
    call print_string

hang:
    hlt
    jmp hang

print_string:
    mov ah, 0x0E
.next:
    lodsb
    test al, al
    jz .done
    int 0x10
    jmp .next
.done:
    ret

hello_msg db "Hello from Stage 2!", 0x0D, 0x0A, 0
